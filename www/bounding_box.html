rangeMapper
A platform for the study of macroecology of life history traits
This script runs with R version 2.15.1 (2012-06-22) & rangeMapper 0.2-0)

SETTING THE SPATIAL BOUNDING BOX

require(rangeMapper)
wd = tempdir()
f = system.file(package = "rangeMapper", "extdata", "wrens", "vector")

Using default values for both bbox and p4s

dbcon = rangeMap.start(file = "test.sqlite", overwrite = TRUE, dir = wd)

##  >> New session 2012-09-12 13:42:33 
##  >> PROJECT: test.sqlite 
## DIRECTORY: C:\Users\valcu\AppData\Local\Temp\RtmpoTziea 

global.bbox.save(con = dbcon)

##  >> Using unprojected global bouging box [ -180,180,-90,90 ]... 
##  >> Bounding box uploaded.
## PROJ4STRING set to 
## NA 

bbox0 = global.bbox.fetch(dbcon)

Using multiple *.shp range files

dbcon = rangeMap.start(file = "test.sqlite", overwrite = TRUE, dir = wd)

##  >> New session 2012-09-12 13:42:33 
##  >> PROJECT: test.sqlite 
## DIRECTORY: C:\Users\valcu\AppData\Local\Temp\RtmpoTziea 

global.bbox.save(con = dbcon, bbox = f)

bbox1 = global.bbox.fetch(dbcon)

Using multiple *.shp range files and reprojecting

dbcon = rangeMap.start(file = "test.sqlite", overwrite = TRUE, dir = wd)

##  >> New session 2012-09-12 13:42:35 
##  >> PROJECT: test.sqlite 
## DIRECTORY: C:\Users\valcu\AppData\Local\Temp\RtmpoTziea 

global.bbox.save(con = dbcon, bbox = f, p4s = CRS("+proj=cea +lon_0=0 +lat_ts=30 +x_0=0 +y_0=0 +ellps=WGS84 +units=m +no_defs "))

##  >> Computing global bounding box for 84 ranges... 
##  >> Checking for proj4 string differences... 
##  >> Done! 
##  >> Converting to  +proj=cea +lon_0=0 +lat_ts=30 +x_0=0 +y_0=0 +ellps=WGS84 +units=m +no_defs 
##  >> Bounding box uploaded.
## PROJ4STRING set to 
##  +proj=cea +lon_0=0 +lat_ts=30 +x_0=0 +y_0=0 +ellps=WGS84 +units=m +no_defs 

bbox2 = global.bbox.fetch(dbcon)

Using a SpatialPolygons object

dbcon = rangeMap.start(file = "test.sqlite", overwrite = TRUE, dir = wd)

##  >> New session 2012-09-12 13:42:37 
##  >> PROJECT: test.sqlite 
## DIRECTORY: C:\Users\valcu\AppData\Local\Temp\RtmpoTziea 


# Troglodytes_aedon
troaed = selectShpFiles(f, ogr = TRUE, polygons.only = TRUE)[71, ]
r = readOGR(troaed$dsn, troaed$layer)

## OGR data source with driver: ESRI Shapefile 
## Source: "C:/Program Files/R/R-2.15.1/library/rangeMapper/extdata/wrens/vector", layer: "Troglodytes_aedon"
## with 1 features and 2 fields
## Feature type: wkbMultiPolygon with 2 dimensions


global.bbox.save(con = dbcon, bbox = r)

##  >> Bounding box uploaded.
## PROJ4STRING set to 
##  +proj=longlat +datum=NAD83 +no_defs +ellps=GRS80 +towgs84=0,0,0 

bbox3 = global.bbox.fetch(dbcon)

par(mfrow = c(2, 2))
plot(bbox0, axes = TRUE)
plot(bbox1, axes = TRUE)
plot(bbox2, axes = TRUE)
plot(bbox3, axes = TRUE)
