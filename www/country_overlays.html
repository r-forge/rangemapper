rangeMapper
A platform for the study of macroecology of life history traits
This script runs with R version 2.15.1 (2012-06-22) & rangeMapper 0.2-3)

Overlay country polygons over a species richness map

# Create a simple rangeMapper project
require(rangeMapper)
dbcon = rangeMap.start(file = "wrens.sqlite", dir = getwd(), overwrite = TRUE)

Breeding range vector files 

loc = system.file(package = "rangeMapper", "extdata", "wrens", "vector_combined")

Mapping using default parameters

global.bbox.save(con = dbcon, bbox = loc)
gridSize.save(dbcon)
canvas.save(dbcon)

Process breeding ranges and create “species_richness” map

r = readOGR(loc, "wrens", verbose = FALSE)
processRanges(spdf = r, con = dbcon, ID = "sci_name")
rangeMap.save(dbcon)

Download Natural Earth polygons

fs = "http://www.naturalearthdata.com/http//www.naturalearthdata.com/download/110m/cultural/ne_110m_admin_0_countries.zip"
if(!file.exists('ne_110m_admin_0_countries.zip')) download.file(fs, "ne_110m_admin_0_countries.zip")
unzip("ne_110m_admin_0_countries.zip")

… read shape file

cntr = readOGR(getwd(), "ne_110m_admin_0_countries")

… and re-project

cntr = spTransform(cntr, CRS(proj4string(global.bbox.fetch(dbcon))))

Plot species richness map and polygons

sr = rangeMap.fetch(dbcon)

Map, countries on top

require(latticeExtra)

spplot(sr, "species_richness", cuts = 20, par.settings = custom.theme(region = brewer.pal(n = 9, 
    name = "YlGnBu")), panel = function(...) {
    panel.levelplot(...)
    sp.polygons(cntr, fill = NA, col = "red")
})
